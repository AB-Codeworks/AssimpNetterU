using System.IO;
using Assimp.Unmanaged;
using UnityEngine;

namespace Assimp {
  /// <summary>
  /// AssimpNet Unity integration. This handles one-time initialization (before scene load) of the AssimpLibrary instance, setting DLL probing paths to load the correct native
  /// dependencies, if the current platform is supported.
  /// </summary>
#if UNITY_EDITOR
    [UnityEditor.InitializeOnLoad]
#endif
  public class AssimpUnity {
    private static bool _triedLoading = false;
    private static bool _assimpAvailable = false;

    /// <summary>
    /// Gets if the assimp library is available on this platform (e.g. the library can load native dependencies).
    /// </summary>
    public static bool IsAssimpAvailable {
      get {
        return _assimpAvailable;
      }
    }

    static AssimpUnity() {
      InitializePlugin();
    }

    [RuntimeInitializeOnLoadMethod(RuntimeInitializeLoadType.BeforeSceneLoad)]
    private static void InitializePlugin() {
      // Only try once during runtime
      if (_triedLoading) {
        Debug.LogWarning($"Already tried loading Assimp for platform: {Application.platform}");
        return;
      }

      UnmanagedLibrary libInstance = AssimpLibrary.Instance;
      libInstance.ThrowOnLoadFailure = true;

      // If already initialized, set flags and return
      if (libInstance.IsLibraryLoaded) {
        _assimpAvailable = true;
        _triedLoading = true;
        Debug.LogWarning($"Assimp was already loaded for {Application.platform}");
        return;
      }

      //First time initialization, need to set a probing path (at least in editor) to resolve the native dependencies
      string pluginsFolder = Path.Combine(Application.dataPath, "Plugins");
      string assimpnetterRoot = Path.Combine(Application.dataPath, "Plugins", "assimpnetter.6.0.4");
      string native64LibPath = null;
      string native32LibPath = null;

      //Set if any platform needs to tweak the default name AssimpNet uses for the platform, null clears using an override at all
      string override64LibName = null;
      string override32LibName = null;

      //Setup DLL paths based on platforms. When run inside the editor, the path will be to the assimpnetter plugin folder structure. When in standalone,
      //Unity copies the native DLLs for the specific target architecture into a single Plugins folder.
      switch (Application.platform) {
        case RuntimePlatform.WindowsEditor:
          native64LibPath = Path.Combine(assimpnetterRoot, "runtimes", "win-x64", "native");
          native32LibPath = Path.Combine(assimpnetterRoot, "runtimes", "win-x86", "native");
          override64LibName = "assimp.dll";
          override32LibName = "assimp.dll";
          break;
        case RuntimePlatform.WindowsPlayer:
          native64LibPath = pluginsFolder;
          native32LibPath = pluginsFolder;
          override64LibName = "assimp.dll";
          override32LibName = "assimp.dll";
          break;
        case RuntimePlatform.LinuxEditor:
          native64LibPath = Path.Combine(assimpnetterRoot, "runtimes", "linux-x64", "native");
          native32LibPath = Path.Combine(assimpnetterRoot, "runtimes", "linux-arm64", "native");
          override64LibName = "libassimp.so";
          override32LibName = "libassimp.so";
          break;
        case RuntimePlatform.LinuxPlayer:
          native64LibPath = pluginsFolder;
          native32LibPath = pluginsFolder;
          override64LibName = "libassimp.so";
          override32LibName = "libassimp.so";
          break;
        case RuntimePlatform.OSXEditor:
          native64LibPath = Path.Combine(assimpnetterRoot, "runtimes", "osx-x64", "native");
          native32LibPath = Path.Combine(assimpnetterRoot, "runtimes", "osx-arm64", "native");
          override64LibName = "libassimp.dylib";
          override32LibName = "libassimp.dylib";
          break;
        case RuntimePlatform.OSXPlayer:
          native64LibPath = pluginsFolder;
          native32LibPath = pluginsFolder;
          override64LibName = "libassimp.dylib";
          override32LibName = "libassimp.dylib";
          break;
          //TODO: Add more platforms if you have binaries that can run on it
      }

      //If both null, then we do not support the platform
      if (native64LibPath == null && native32LibPath == null) {
        Debug.Log($"AssimpNetter does not support platform: {Application.platform}");
        _assimpAvailable = false;
        _triedLoading = true;
        return;
      }

      //Set resolver properties, null will clear the property
      libInstance.Resolver.SetOverrideLibraryName64(override64LibName);
      libInstance.Resolver.SetOverrideLibraryName32(override32LibName);
      libInstance.Resolver.SetProbingPaths64(native64LibPath);
      libInstance.Resolver.SetProbingPaths32(native32LibPath);

      //Try and load the native library at domain reload
      _assimpAvailable = libInstance.LoadLibrary();
      _triedLoading = true;
      Debug.Log("Successfully loaded AssimpNetter library!");
    }
  }
}
